#include "EventHandler.h"
#include "FontManager.h"
#include "Image.h"
#include "Line.h"
#include "Point.h"
#include "Rectangle.h"
#include "Renderer.h"
#include "SDLSystem.h"
#include "TextField.h"
#include "Window.h"

#include <cstdlib>
#include <ctime>
#include <iostream>
#include <vector>

int pauseTime = 300;

std::string text1 = "Hello, World!";
std::string text2 = "Hello, Planet!";

void moveRock(SDL_Rect& rockSrcRect, int windowWidth, int windowHeight, int rockSpeed)
{
    if (rockSrcRect.y > windowHeight)
    {
        rockSrcRect.y = -10;
        rockSrcRect.x = rand() % (windowWidth - rockSrcRect.w);
    }
    rockSrcRect.y += rockSpeed;
}

bool checkCollision(const SDL_Rect& object1, const SDL_Rect& object2)
{
    return SDL_HasIntersection(&object1, &object2);
}

int main()
{
    SDLSystem sdl(SDL_INIT_VIDEO);
    Window    window("SDL2", SDL_WINDOWPOS_CENTERED, SDL_WINDOWPOS_CENTERED, 640, 480, SDL_WINDOW_SHOWN);
    Renderer  renderer(window.getWindow(), -1, SDL_RENDERER_ACCELERATED);
    renderer.setRenderDrawColor(255, 255, 255, 255);

    auto      font = FontManager::loadFont("fonts/LiberationSans-Bold.ttf", 24);
    TextField textField(renderer.getRenderer(), text1, { 0, 0, 200, 50 }, { 0, 0, 0, 255 }, font.get());

    Image    person(renderer.getRenderer(), "pngs/Person_3.png");
    SDL_Rect personSrcRect = { 220, 205, 70, 200 };
    person.setDestRect(personSrcRect);

    Image    rock(renderer.getRenderer(), "pngs/Rock.png");
    SDL_Rect rockSrcRect = { 100, -10, 50, 50 };
    rock.setDestRect(rockSrcRect);

    Image    rock2(renderer.getRenderer(), "pngs/Rock.png");
    SDL_Rect rock2SrcRect = { 300, -50, 50, 50 };
    rock2.setDestRect(rock2SrcRect);

    std::vector<SDL_Point> points = { { 50, 100 }, { 100, 150 }, { 150, 200 }, { 200, 250 } };
    Point                  point(renderer.getRenderer(), points);
    point.addPoint({ 250, 300 });

    Line line(renderer.getRenderer(), { 0, 0 }, { 640, 480 });
    line.addLine({ 640, 0 }, { 0, 480 });

    std::vector<std::pair<SDL_Point, SDL_Point>> lines = { { { 50, 0 }, { 50, 480 } }, { { 100, 0 }, { 100, 480 } } };
    Line                                         line2(renderer.getRenderer(), lines);

    SDL_Rect  rect1 = { 50, 50, 100, 100 };
    SDL_Rect  rect2 = { 200, 50, 100, 100 };
    Rectangle filledRect(renderer.getRenderer(), rect1);
    filledRect.addRect(rect2);
    Rectangle outlinedRect(renderer.getRenderer(), { 350, 50, 100, 100 });

    SDL_Rect rect = { 270, 190, 100, 100 };

    EventHandler eventHandler;
    srand(static_cast<unsigned int>(time(nullptr)));

    bool running = true;
    while (running)
    {
        while (eventHandler.pollEvent())
        {
            if (eventHandler.isQuit())
            {
                running = false;
            }
            else if (eventHandler.isKeyDown())
            {
                switch (eventHandler.getKeyCode())
                {
                    case SDLK_UP:
                        personSrcRect.y -= 10;
                        break;
                    case SDLK_DOWN:
                        personSrcRect.y += 10;
                        break;
                    case SDLK_LEFT:
                        personSrcRect.x -= 10;
                        break;
                    case SDLK_RIGHT:
                        personSrcRect.x += 10;
                        break;
                }
                person.setDestRect(personSrcRect);
            }
        }
        moveRock(rockSrcRect, 640, 480, 3);
        moveRock(rock2SrcRect, 640, 480, 3);

        if (checkCollision(personSrcRect, rockSrcRect) || checkCollision(personSrcRect, rock2SrcRect))
        {
            std::cout << "Collision detected! Exiting game." << std::endl;
            running = false;
        }

        renderer.clear();
        person.render();
        rock.render();
        rock2.render();
        renderer.present();
        SDL_Delay(16);
    }

    return 0;
}
